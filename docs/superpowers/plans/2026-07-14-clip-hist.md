# clip-hist Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build clip-hist — a macOS clipboard-history recorder (Swift watcher + launchd) with a terminal fzf picker bound to Ctrl+H in zsh, an 8-hour default retention window, and a small CLI that doubles as the agent interface.

**Architecture:** A ~90-line Swift binary polls `NSPasteboard.changeCount` every 0.3s and appends text items to `~/.local/share/clip-hist/history.jsonl` (dedupe, retention prune, caps). A bash CLI (`bin/clip-hist`) reads/manages the store using embedded `/usr/bin/python3` for JSON. A zsh ZLE widget pipes the history through fzf and inserts the selection at the cursor.

**Tech Stack:** Swift (AppKit, compiled with `swiftc`, no Xcode project), bash (macOS 3.2-compatible), /usr/bin/python3 (ships with Command Line Tools), zsh/ZLE, fzf, launchd.

**Spec:** `docs/superpowers/specs/2026-07-14-clip-hist-design.md`

## Global Constraints

- macOS only. Repo lives at `the repo root`. License: MIT.
- Data dir `~/.local/share/clip-hist/` (override via `CLIPHIST_DATA_DIR` env var — required for tests): `history.jsonl` (mode 600), `config.json`, `paused` sentinel.
- Retention default **8h**; accepted values `Nm`/`Nh`/`Nd`/`off`. Backstop caps: **500 items**, **100 KB per item**.
- Watcher skips: `org.nspasteboard.ConcealedType`, `org.nspasteboard.TransientType`, empty/whitespace-only, non-text, consecutive duplicates, anything while `paused` exists.
- launchd label: `com.rajsingh.clip-hist`. Watcher binary: `~/.local/bin/clip-hist-watcher`. CLI symlink: `~/.local/bin/clip-hist`.
- Default keybinding `^H`, overridable via `CLIPHIST_KEY` exported before sourcing `shell/clip-hist.zsh`.
- Bash scripts must run on macOS's default bash 3.2 (no associative arrays, no `${var,,}`).
- **Tests are session-only** (Raj's global rule): they live in `tests/`, which is gitignored — never commit them. Everything else commits normally.
- Commits: plain messages, no co-author or attribution lines.

```bash
#!/bin/bash
set -euo pipefail

DATA_DIR="${CLIPHIST_DATA_DIR:-$HOME/.local/share/clip-hist}"
HISTORY_FILE="$DATA_DIR/history.jsonl"
CONFIG_FILE="$DATA_DIR/config.json"
PAUSE_FILE="$DATA_DIR/paused"
DEFAULT_RETENTION="8h"
PY=/usr/bin/python3

usage() {
  cat <<'EOF'
clip-hist — macOS clipboard history

Usage:
  clip-hist list [-n N] [--json]   Show history, newest first
  clip-hist get INDEX              Print raw text of item INDEX (from list)
  clip-hist retention [DURATION]   Show or set retention (e.g. 30m, 8h, 2d, off)
  clip-hist key [BINDING]          Show or set picker keybinding (^K or ctrl-k)
  clip-hist clear [--force]        Delete all history
  clip-hist pause | resume         Pause/resume recording
  clip-hist status                 Watcher/paused state, retention, item count
EOF
}

duration_to_seconds() {
  local d="$1"
  if [[ "$d" == "off" ]]; then echo off; return 0; fi
  if [[ "$d" =~ ^([0-9]+)([mhd])$ ]]; then
    local n="${BASH_REMATCH[1]}" u="${BASH_REMATCH[2]}"
    case "$u" in
      m) echo $((n * 60)) ;;
      h) echo $((n * 3600)) ;;
      d) echo $((n * 86400)) ;;
    esac
    return 0
  fi
  return 1
}

get_retention() {
  if [[ -f "$CONFIG_FILE" ]]; then
    $PY -c 'import json,sys; print(json.load(open(sys.argv[1])).get("retention","8h"))' \
      "$CONFIG_FILE" 2>/dev/null || echo "$DEFAULT_RETENTION"
  else
    echo "$DEFAULT_RETENTION"
  fi
}

retention_cutoff() {
  local secs
  secs=$(duration_to_seconds "$(get_retention)") || secs=$(duration_to_seconds "$DEFAULT_RETENTION")
  if [[ "$secs" == "off" ]]; then
    echo 0
  else
    echo $(( $(date +%s) - secs ))
  fi
}

set_config() {
  # set_config KEY VALUE — merge into config.json (never clobber other keys)
  mkdir -p "$DATA_DIR"
  $PY -c '
import json, sys
path, k, v = sys.argv[1], sys.argv[2], sys.argv[3]
cfg = {}
try:
    cfg = json.load(open(path))
except Exception:
    pass
cfg[k] = v
json.dump(cfg, open(path, "w"))
' "$CONFIG_FILE" "$1" "$2"
  chmod 600 "$CONFIG_FILE"
}

cmd_retention() {
  if [[ $# -eq 0 ]]; then get_retention; return 0; fi
  local d="$1"
  if ! duration_to_seconds "$d" >/dev/null; then
    echo "invalid duration: $d (use e.g. 30m, 8h, 2d, off)" >&2
    exit 1
  fi
  set_config retention "$d"
  echo "retention set to $d"
}

cmd="${1:-}"
if [[ $# -gt 0 ]]; then shift; fi
case "$cmd" in
  retention) cmd_retention "$@" ;;
  ""|help|-h|--help) usage ;;
  *) echo "unknown command: $cmd" >&2; usage >&2; exit 1 ;;
esac
```

```bash
render() {
  # render MODE N [INDEX] — MODE: json|human|feed|get
  $PY - "$HISTORY_FILE" "$(retention_cutoff)" "$@" <<'PYEOF'
import json, sys, time

path, cutoff, mode = sys.argv[1], float(sys.argv[2]), sys.argv[3]
n = int(sys.argv[4]) if len(sys.argv) > 4 else 0
idx = int(sys.argv[5]) if len(sys.argv) > 5 else -1

items = []
try:
    with open(path) as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            try:
                o = json.loads(line)
            except ValueError:
                continue
            if not isinstance(o, dict) or not o.get("text"):
                continue
            if float(o.get("ts", 0)) < cutoff:
                continue
            items.append(o)
except FileNotFoundError:
    pass

items.reverse()  # newest first

if mode == "get":
    if 0 <= idx < len(items):
        sys.stdout.write(items[idx]["text"])
        sys.exit(0)
    sys.exit(1)

if n > 0:
    items = items[:n]

def rel(ts):
    d = int(time.time() - ts)
    if d < 60: return "%ds" % d
    if d < 3600: return "%dm" % (d // 60)
    if d < 86400: return "%dh" % (d // 3600)
    return "%dd" % (d // 86400)

for i, o in enumerate(items):
    if mode == "json":
        print(json.dumps(o))
        continue
    flat = o["text"].replace("\t", " ").replace("\r", "").replace("\n", "⏎")
    app = str(o.get("app", "?")).split(".")[-1]
    if mode == "feed":
        print("%d\t%4s %-12s %s" % (i, rel(float(o["ts"])), app[:12], flat))
    else:
        if len(flat) > 100:
            flat = flat[:100] + "…"
        print("%3d  %4s  %-16s %s" % (i, rel(float(o["ts"])), app[:16], flat))
PYEOF
}

cmd_list() {
  local n=0 mode=human
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --json) mode=json; shift ;;
      -n) n="${2:-0}"; shift 2 ;;
      *) echo "unknown flag: $1" >&2; exit 1 ;;
    esac
  done
  render "$mode" "$n"
}
```

```swift
    ?? fm.homeDirectoryForCurrentUser.appendingPathComponent(".local/share/clip-hist")
let historyURL = dataDir.appendingPathComponent("history.jsonl")
let configURL = dataDir.appendingPathComponent("config.json")
let pauseURL = dataDir.appendingPathComponent("paused")

let maxItems = 500
let maxItemBytes = 100 * 1024
let pollInterval: TimeInterval = 0.3
let defaultRetention: TimeInterval = 8 * 3600

let concealedType = NSPasteboard.PasteboardType("org.nspasteboard.ConcealedType")
let transientType = NSPasteboard.PasteboardType("org.nspasteboard.TransientType")

func parseObject(_ line: String) -> [String: Any]? {
    guard let d = line.data(using: .utf8),
          let o = try? JSONSerialization.jsonObject(with: d) as? [String: Any] else { return nil }
    return o
}

// nil means "off" (keep forever)
func retentionSeconds() -> TimeInterval? {
    guard let data = try? Data(contentsOf: configURL),
          let obj = try? JSONSerialization.jsonObject(with: data) as? [String: Any],
          let r = obj["retention"] as? String else { return defaultRetention }
    if r == "off" { return nil }
    guard r.count >= 2, let n = Double(r.dropLast()) else { return defaultRetention }
    switch r.last! {
    case "m": return n * 60
    case "h": return n * 3600
    case "d": return n * 86400
    default: return defaultRetention
    }
}

func loadLines() -> [String] {
    guard let s = try? String(contentsOf: historyURL, encoding: .utf8) else { return [] }
    return s.split(separator: "\n").map(String.init)
}

func append(text: String, app: String) {
    var lines = loadLines()
    if let last = lines.last, parseObject(last)?["text"] as? String == text { return }

    let record: [String: Any] = ["ts": Date().timeIntervalSince1970, "app": app, "text": text]
    guard let d = try? JSONSerialization.data(withJSONObject: record),
          let json = String(data: d, encoding: .utf8) else { return }
    lines.append(json)

    if let ret = retentionSeconds() {
        let cutoff = Date().timeIntervalSince1970 - ret
        lines = lines.filter { line in
            guard let ts = parseObject(line)?["ts"] as? Double else { return false }
            return ts >= cutoff
        }
    }
    if lines.count > maxItems { lines = Array(lines.suffix(maxItems)) }

    try? fm.createDirectory(at: dataDir, withIntermediateDirectories: true)
    try? (lines.joined(separator: "\n") + "\n")
        .write(to: historyURL, atomically: true, encoding: .utf8)
    try? fm.setAttributes([.posixPermissions: 0o600], ofItemAtPath: historyURL.path)
}

let pb = NSPasteboard.general
var lastChange = pb.changeCount

while true {
    Thread.sleep(forTimeInterval: pollInterval)
    guard pb.changeCount != lastChange else { continue }
    lastChange = pb.changeCount

    if fm.fileExists(atPath: pauseURL.path) { continue }
    let types = pb.types ?? []
    if types.contains(concealedType) || types.contains(transientType) { continue }
    guard let text = pb.string(forType: .string) else { continue }
    if text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty { continue }
    if text.utf8.count > maxItemBytes { continue }

    let front = NSWorkspace.shared.frontmostApplication
    let app = front?.bundleIdentifier ?? front?.localizedName ?? "unknown"
    append(text: text, app: app)
}
```

### Task 6: zsh widget + fzf picker

**Files:**
- Create: `shell/clip-hist.zsh`

**Interfaces:**
- Consumes: `clip-hist pick-feed`, `clip-hist get INDEX` (Task 3), `clip-hist key` (Task 4)
- Produces: ZLE widget `_clip-hist-pick` bound at source time — `CLIPHIST_KEY` env var if set, else `clip-hist key` (config value, default `^H`); Enter inserts at cursor, Ctrl+Y re-copies; ensures `~/.local/bin` on PATH

- [ ] **Step 1: Write `shell/clip-hist.zsh`**

```zsh
# clip-hist — clipboard history picker for zsh.
# Sourced from .zshrc by install.sh. Keybinding resolution (first wins):
# CLIPHIST_KEY env var > `clip-hist key` config value > ^H.
# Change persistently with: clip-hist key ctrl-k

[[ ":$PATH:" == *":$HOME/.local/bin:"* ]] || export PATH="$PATH:$HOME/.local/bin"

_clip-hist-pick() {
  emulate -L zsh
  setopt local_options pipefail
  if ! command -v clip-hist >/dev/null 2>&1; then
    zle -M "clip-hist: CLI not found in PATH"
    return 1
  fi
  if ! command -v fzf >/dev/null 2>&1; then
    zle -M "clip-hist: fzf not installed — brew install fzf"
    return 1
  fi
  local out key line idx text
  out=$(clip-hist pick-feed | fzf --delimiter=$'\t' --with-nth=2 \
        --height=40% --reverse --no-multi --expect=ctrl-y \
        --prompt='clipboard> ') || { zle reset-prompt; return 0 }
  key=${out%%$'\n'*}
  line=${out#*$'\n'}
  idx=${line%%$'\t'*}
  text=$(clip-hist get "$idx") || { zle reset-prompt; return 1 }
  if [[ $key == ctrl-y ]]; then
    print -rn -- "$text" | pbcopy
    zle -M "clip-hist: copied to clipboard"
  else
    LBUFFER+="$text"
  fi
  zle reset-prompt
}
zle -N _clip-hist-pick
_clip_hist_key="${CLIPHIST_KEY:-$(command clip-hist key 2>/dev/null || echo '^H')}"
bindkey "$_clip_hist_key" _clip-hist-pick
unset _clip_hist_key
```

```bash
#!/bin/bash
set -euo pipefail

REPO_DIR="$(cd "$(dirname "$0")" && pwd)"
BIN_DIR="$HOME/.local/bin"
DATA_DIR="$HOME/.local/share/clip-hist"
PLIST="$HOME/Library/LaunchAgents/com.rajsingh.clip-hist.plist"

  echo "swiftc not found. Install Command Line Tools first: xcode-select --install" >&2
  exit 1

if ! command -v fzf >/dev/null 2>&1; then
  if command -v brew >/dev/null 2>&1; then
    echo "Installing fzf..."
    brew install fzf
  else
    echo "warning: fzf not found and Homebrew missing — the Ctrl+H picker needs fzf" >&2
  fi
fi

mkdir -p "$BIN_DIR" "$DATA_DIR" "$HOME/Library/LaunchAgents"
chmod 700 "$DATA_DIR"

echo "Compiling watcher..."
swiftc -O "$REPO_DIR/watcher/main.swift" -o "$BIN_DIR/clip-hist-watcher"

ln -sf "$REPO_DIR/bin/clip-hist" "$BIN_DIR/clip-hist"

cat > "$PLIST" <<EOF
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>Label</key><string>com.rajsingh.clip-hist</string>
  <key>ProgramArguments</key>
  <array><string>$BIN_DIR/clip-hist-watcher</string></array>
  <key>RunAtLoad</key><true/>
  <key>KeepAlive</key><true/>
  <key>StandardErrorPath</key><string>$DATA_DIR/watcher.log</string>
</dict>
</plist>
EOF

launchctl unload "$PLIST" 2>/dev/null || true
launchctl load "$PLIST"

ZLINE="source \"$REPO_DIR/shell/clip-hist.zsh\""
if ! grep -qF "$ZLINE" "$HOME/.zshrc" 2>/dev/null; then
  printf '\n%s\n' "$ZLINE" >> "$HOME/.zshrc"
  echo "Added source line to ~/.zshrc"
fi

echo
echo "clip-hist installed. Open a new terminal (or run: source ~/.zshrc), copy"
echo "something, then press Ctrl+H. Check health any time with: clip-hist status"
```
